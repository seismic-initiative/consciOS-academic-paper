% Preprint font configuration (serif body, bold headings)
% Body: TeX Gyre Termes (Times-like); Headings: bold serif
\usepackage{fontspec}

% Serif body font (bundled with TeX):
\IfFontExistsTF{TeX Gyre Termes}{\setmainfont{TeX Gyre Termes}}{\setmainfont{Times New Roman}}

% Bold headings in serif with increased size - use titlesec for comprehensive control
\usepackage{titlesec}

% Format all heading levels to be bold serif with appropriate sizes
\titleformat{\section}{\Large\bfseries\rmfamily}{\thesection}{1em}{}
\titleformat{\subsection}{\large\bfseries\rmfamily}{\thesubsection}{1em}{}
\titleformat{\subsubsection}{\normalsize\bfseries\rmfamily}{\thesubsubsection}{1em}{}
\titleformat{\paragraph}[block]{\normalsize\bfseries\rmfamily}{\theparagraph}{0.5em}{}
\titleformat{\subparagraph}[block]{\normalsize\bfseries\rmfamily}{\thesubparagraph}{0.5em}{}

% More spacing around all heading levels (before and after)
\titlespacing*{\section}{0pt}{2ex plus 0.5ex minus 0.2ex}{1.5ex plus 0.2ex}
\titlespacing*{\subsection}{0pt}{1.8ex plus 0.5ex minus 0.2ex}{1.2ex plus 0.2ex}
\titlespacing*{\subsubsection}{0pt}{1.5ex plus 0.5ex minus 0.2ex}{1ex plus 0.2ex}
\titlespacing*{\paragraph}{0pt}{1.2ex plus 0.5ex minus 0.2ex}{0.8ex plus 0.2ex}
\titlespacing*{\subparagraph}{0pt}{1ex plus 0.5ex minus 0.2ex}{0.7ex plus 0.2ex}

% Title styling
\usepackage{titling}
\pretitle{\begin{center}\bfseries\rmfamily\Huge}
\posttitle{\par\end{center}}
